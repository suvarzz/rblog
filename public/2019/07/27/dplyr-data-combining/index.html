Dplyr data combining

library(dplyr)
a = data.frame(x1=c('A','B','C'), x2=c(1,2,3))
b = data.frame(x1=c('A','B','D'), x3=c('T','F','T'))
a
##   x1 x2
## 1  A  1
## 2  B  2
## 3  C  3
b
##   x1 x3
## 1  A  T
## 2  B  F
## 3  D  T
left_join(a, b, by='x1')
## Warning: Column `x1` joining factors with different levels, coercing to
## character vector
##   x1 x2   x3
## 1  A  1    T
## 2  B  2    F
## 3  C  3 <NA>
